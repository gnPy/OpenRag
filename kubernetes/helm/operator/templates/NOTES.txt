To check the operator logs:

  kubectl logs -n {{ .Release.Namespace }} -l control-plane=controller-manager -f

The operator will watch for OpenRAG custom resources. To create an OpenRAG instance:

  kubectl apply -f - <<EOF
  apiVersion: openr.ag/v1alpha1
  kind: OpenRAG
  metadata:
    name: my-openrag
    namespace: default
  spec:
    frontend:
      image: langflowai/openrag-frontend:latest
    backend:
      image: langflowai/openrag-backend:latest
      envSecret: my-backend-env
      storage:
        enabled: true
        size: 10Gi
    langflow:
      image: langflowai/openrag-langflow:latest
      envSecret: my-langflow-env
      storage:
        enabled: true
        size: 10Gi
    opensearch:
      host: opensearch.default.svc.cluster.local
      credentialsSecret: opensearch-credentials
  EOF
